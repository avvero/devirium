# Выгрузка комитов сделанных в ветку

```bash
#!/bin/bash

START_DATE="2023-09-15"

git checkout master

# Получение списка хешей коммитов слияния после указанной даты
merge_commits=$(git log --merges --after="$START_DATE" --pretty=format:"%H")

echo "--------------------------------"
# Перебор каждого коммита слияния
for commit in $merge_commits; do
#    echo "Merge commit: $commit"

    # Получение родительских коммитов для коммита слияния
    parents=$(git show -s --pretty=format:"%P" $commit)

    # Перебор родительских коммитов
    for parent in $parents; do
#        echo "Parent commit: $parent"
        # Вывод сообщения последнего коммита перед слиянием, без использования пейджера
        message=$(git --no-pager log -1 $parent --pretty=format:"%H - %ad - %s" --date=format:"%Y-%m-%d")

        if ! [[ $message =~ Merge\ branch ]]; then
            # Вывод хеша и сообщения коммита
            echo $message
        fi
    done
done
echo "--------------------------------"
```
